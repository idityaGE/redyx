Alternative: Use an init container that copies from a built image.

**deploy/k8s/charts/redyx-services/templates/envoy-deployment.yaml:**
```yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: envoy
  namespace: {{ .Release.Namespace }}
spec:
  replicas: 1
  selector:
    matchLabels:
      app.kubernetes.io/name: envoy
  template:
    metadata:
      labels:
        app.kubernetes.io/name: envoy
    spec:
      containers:
        - name: envoy
          image: envoyproxy/envoy:v1.37.0
          ports:
            - containerPort: 8080
              name: http
            - containerPort: 9901
              name: admin
          volumeMounts:
            - name: config
              mountPath: /etc/envoy/envoy.yaml
              subPath: envoy.yaml
            - name: proto
              mountPath: /etc/envoy/proto.pb
              subPath: proto.pb
          readinessProbe:
            httpGet:
              path: /ready
              port: 9901
            initialDelaySeconds: 5
            periodSeconds: 10
          livenessProbe:
            httpGet:
              path: /ready
              port: 9901
            initialDelaySeconds: 10
            periodSeconds: 20
          resources:
            requests:
              cpu: 100m
              memory: 128Mi
            limits:
              cpu: 500m
              memory: 512Mi
      volumes:
        - name: config
          configMap:
            name: envoy-config
        - name: proto
          configMap:
            name: envoy-proto
---
apiVersion: v1
kind: Service
metadata:
  name: envoy
  namespace: {{ .Release.Namespace }}
spec:
  type: NodePort  # External access
  ports:
    - port: 8080
      targetPort: 8080
      nodePort: 30080
      name: http
    - port: 9901
      targetPort: 9901
      name: admin
  selector:
    app.kubernetes.io/name: envoy
```
