    ```bash
    # Register user
    curl -s http://localhost:8080/api/v1/auth/register -d '{"email":"test@test.com","username":"testuser","password":"Test1234!"}' | jq

    # Login (skip OTP for dev — check if auto-verified)
    curl -s http://localhost:8080/api/v1/auth/login -d '{"email":"test@test.com","password":"Test1234!"}' | jq
    # Save accessToken from response
    
    # Create community
    curl -s http://localhost:8080/api/v1/communities -H "Authorization: Bearer $TOKEN" -d '{"name":"testcommunity","description":"Test community"}' | jq
    ```

    **Post CRUD tests:**
    ```bash
    # 1. Create text post (POST-01)
    curl -s http://localhost:8080/api/v1/posts -H "Authorization: Bearer $TOKEN" \
      -d '{"communityName":"testcommunity","title":"First test post","body":"Hello **world** from markdown!","postType":"POST_TYPE_TEXT"}' | jq
    # Save postId from response

    # 2. Create link post (POST-02)
    curl -s http://localhost:8080/api/v1/posts -H "Authorization: Bearer $TOKEN" \
      -d '{"communityName":"testcommunity","title":"Cool link","url":"https://github.com/redyx","postType":"POST_TYPE_LINK"}' | jq

    # 3. Create anonymous post (POST-08)
    curl -s http://localhost:8080/api/v1/posts -H "Authorization: Bearer $TOKEN" \
      -d '{"communityName":"testcommunity","title":"Anonymous thought","body":"You cant see me","postType":"POST_TYPE_TEXT","isAnonymous":true}' | jq
    # Verify: authorUsername should be "[anonymous]", authorId should be ""

    # 4. Get post (POST-04)
    curl -s http://localhost:8080/api/v1/posts/$POST_ID -H "Authorization: Bearer $TOKEN" | jq
    # Verify: post fields, userVote=0, isSaved=false

    # 5. Update post (POST-05)
    curl -s -X PATCH http://localhost:8080/api/v1/posts/$POST_ID -H "Authorization: Bearer $TOKEN" \
      -d '{"body":"Updated body content"}' | jq
    # Verify: isEdited=true, editedAt set

    # 6. Media post returns error (POST-03 stub)
    curl -s http://localhost:8080/api/v1/posts -H "Authorization: Bearer $TOKEN" \
      -d '{"communityName":"testcommunity","title":"Media test","postType":"POST_TYPE_MEDIA"}' | jq
    # Verify: returns UNIMPLEMENTED or appropriate error
    ```

    **Feed tests:**
    ```bash

      -H "Authorization: Bearer $TOKEN" | jq

    # 8. Community feed - New sort
    curl -s "http://localhost:8080/api/v1/communities/testcommunity/posts?sort=SORT_ORDER_NEW" \
      -H "Authorization: Bearer $TOKEN" | jq

    # 9. Community feed - Top sort with time range
    curl -s "http://localhost:8080/api/v1/communities/testcommunity/posts?sort=SORT_ORDER_TOP&timeRange=TIME_RANGE_DAY" \
      -H "Authorization: Bearer $TOKEN" | jq

    # 10. Home feed (POST-07)
    curl -s "http://localhost:8080/api/v1/feed?sort=SORT_ORDER_HOT" -H "Authorization: Bearer $TOKEN" | jq
    # Verify: returns posts from testcommunity (user is member)
    ```

    **Vote tests:**
    ```bash
    # 11. Upvote (VOTE-01)
    curl -s http://localhost:8080/api/v1/votes -H "Authorization: Bearer $TOKEN" \
      -d '{"targetId":"'$POST_ID'","targetType":"TARGET_TYPE_POST","direction":"VOTE_DIRECTION_UP"}' | jq
    # Verify: newScore = 1

    # 12. Change to downvote (VOTE-02)
    curl -s http://localhost:8080/api/v1/votes -H "Authorization: Bearer $TOKEN" \
      -d '{"targetId":"'$POST_ID'","targetType":"TARGET_TYPE_POST","direction":"VOTE_DIRECTION_DOWN"}' | jq
    # Verify: newScore = -1

    # 13. Remove vote
    curl -s http://localhost:8080/api/v1/votes -H "Authorization: Bearer $TOKEN" \
      -d '{"targetId":"'$POST_ID'","targetType":"TARGET_TYPE_POST","direction":"VOTE_DIRECTION_NONE"}' | jq
    # Verify: newScore = 0

    # 14. Idempotent re-vote (VOTE-05)
    curl -s http://localhost:8080/api/v1/votes -H "Authorization: Bearer $TOKEN" \
      -d '{"targetId":"'$POST_ID'","targetType":"TARGET_TYPE_POST","direction":"VOTE_DIRECTION_UP"}' | jq
    curl -s http://localhost:8080/api/v1/votes -H "Authorization: Bearer $TOKEN" \
      -d '{"targetId":"'$POST_ID'","targetType":"TARGET_TYPE_POST","direction":"VOTE_DIRECTION_UP"}' | jq
    # Verify: both return same newScore = 1

    # 15. Get vote state
    curl -s http://localhost:8080/api/v1/votes/TARGET_TYPE_POST/$POST_ID -H "Authorization: Bearer $TOKEN" | jq
    # Verify: direction = VOTE_DIRECTION_UP
    ```

    **Save tests:**
    ```bash
    # 16. Save post (POST-09)
    curl -s http://localhost:8080/api/v1/posts/$POST_ID/save -H "Authorization: Bearer $TOKEN" \

    # 17. List saved posts
    curl -s http://localhost:8080/api/v1/saved -H "Authorization: Bearer $TOKEN" | jq

    # 18. Unsave
    curl -s http://localhost:8080/api/v1/posts/$POST_ID/save -H "Authorization: Bearer $TOKEN" \
      -d '{"save":false}' | jq

    # 19. List saved again — should be empty
    curl -s http://localhost:8080/api/v1/saved -H "Authorization: Bearer $TOKEN" | jq
    ```

    ```bash

    curl -s http://localhost:8080/api/v1/posts/$POST_ID | jq
    # Verify: isDeleted=true, title/body show [deleted]
    ```
