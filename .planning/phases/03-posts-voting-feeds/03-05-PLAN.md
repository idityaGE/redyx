<!-- API shapes -->

Create post: POST /api/v1/posts
  Body: { communityName, title, body, url, postType, isAnonymous }
  Response: { post: Post }

Get post: GET /api/v1/posts/{post_id}
  Response: { post: Post, userVote: number, isSaved: boolean }

Update post: PATCH /api/v1/posts/{post_id}
  Body: { title, body, url }
  Response: { post: Post }

Delete post: DELETE /api/v1/posts/{post_id}
  Response: {}

List community posts: GET /api/v1/communities/{name}/posts?sort=&timeRange=&pagination.limit=&pagination.cursor=
  Response: { posts: Post[], pagination: { nextCursor, hasMore } }

List home feed: GET /api/v1/feed?sort=&timeRange=&pagination.limit=&pagination.cursor=
  Response: { posts: Post[], pagination: { nextCursor, hasMore } }

Save/unsave: POST /api/v1/posts/{post_id}/save
  Body: { save: boolean }
  Response: {}
